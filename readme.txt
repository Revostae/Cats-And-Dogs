Add this under mechSizeMultipliers: (Make sure to add a coma to the one above it!)
			"chassisdef_madcatzeropointfive_MC-05": 0.72,
			"chassisdef_madcatIII_MCIII_1": 0.85,
			"chassisdef_madcatIII_MCIII_2": 0.85,
			"chassisdef_madcatIII_MCIII_3": 0.85,
			"chassisdef_madcatIII_MCIII_4": 0.85,
			"chassisdef_madcatIII_MCIII_5": 0.85
